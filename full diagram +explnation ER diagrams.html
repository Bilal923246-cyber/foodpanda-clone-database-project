👤 USERS → CUSTOMERS One‑to‑one: one user account is linked to exactly one customer profile.
👤 USERS → RIDERS One‑to‑one: one user account is linked to exactly one rider profile.
🛍️ CUSTOMER → ORDER One‑to‑many: a single customer can place many orders over time.
🏪 RESTAURANT → MENU_ITEM One‑to‑many: one restaurant offers many menu items.
📦 ORDER → ORDER_ITEMS One‑to‑many: one order contains multiple line items.
💳 ORDER → PAYMENT One‑to‑one: each order has exactly one payment record.
🛵 RIDER → DELIVERY One‑to‑many: one rider can perform many deliveries (assignments).
⭐ CUSTOMER → REVIEW One‑to‑many: a customer can write many reviews for different orders.
🍽️ RESTAURANT → REVIEW One‑to‑many: a restaurant can receive many reviews from different customers.

🔍 Normalization (3NF) – Explanation

  • 1NF: No repeating groups – e.g., Order_Items has one row per item, not a comma list.
  • 2NF: No partial dependency – unit_price in Order_Items is a snapshot, not dependent on order_id.
  • 3NF: No transitive dependency – Orders stores customer_id, not customer name/address (joined when needed).

✅ All 22 tables satisfy 3NF, eliminating update/insert/delete anomalies.

CREATE DATABASE FoodPandaClone;
GO
USE FoodPandaClone;
GO

-- 1. Users
CREATE TABLE Users (
    user_id INT IDENTITY PRIMARY KEY,
    email NVARCHAR(100) UNIQUE NOT NULL,
    password_hash NVARCHAR(255) NOT NULL,
    user_type NVARCHAR(20) CHECK (user_type IN ('customer','owner','rider'))
);

-- 2. Customers
CREATE TABLE Customers (
    customer_id INT IDENTITY PRIMARY KEY,
    user_id INT UNIQUE FOREIGN KEY REFERENCES Users(user_id) ON DELETE CASCADE,
    first_name NVARCHAR(50),
    loyalty_points INT DEFAULT 0
);

-- 3. Restaurant_Owners
CREATE TABLE Restaurant_Owners (
    owner_id INT IDENTITY PRIMARY KEY,
    user_id INT UNIQUE FOREIGN KEY REFERENCES Users(user_id),
    business_license NVARCHAR(50)
);

-- 4. Riders
CREATE TABLE Riders (
    rider_id INT IDENTITY PRIMARY KEY,
    user_id INT UNIQUE FOREIGN KEY REFERENCES Users(user_id),
    first_name NVARCHAR(50),
    rating DECIMAL(3,2) DEFAULT 5.0
);

-- 5. Addresses
CREATE TABLE Addresses (
    address_id INT IDENTITY PRIMARY KEY,
    address_line NVARCHAR(255),
    city NVARCHAR(50)
);

-- 6. Restaurants
CREATE TABLE Restaurants (
    restaurant_id INT IDENTITY PRIMARY KEY,
    owner_id INT FOREIGN KEY REFERENCES Restaurant_Owners(owner_id),
    name NVARCHAR(100)
);

-- 7. Menu_Items
CREATE TABLE Menu_Items (
    item_id INT IDENTITY PRIMARY KEY,
    restaurant_id INT FOREIGN KEY REFERENCES Restaurants(restaurant_id),
    name NVARCHAR(100),
    price DECIMAL(10,2)
);

-- 8. Orders
CREATE TABLE Orders (
    order_id INT IDENTITY PRIMARY KEY,
    customer_id INT FOREIGN KEY REFERENCES Customers(customer_id),
    restaurant_id INT FOREIGN KEY REFERENCES Restaurants(restaurant_id),
    total_amount DECIMAL(10,2),
    status NVARCHAR(20) DEFAULT 'Pending'
);

-- 9. Order_Items (weak entity)
CREATE TABLE Order_Items (
    order_item_id INT IDENTITY PRIMARY KEY,
    order_id INT FOREIGN KEY REFERENCES Orders(order_id) ON DELETE CASCADE,
    item_id INT FOREIGN KEY REFERENCES Menu_Items(item_id),
    quantity INT,
    unit_price DECIMAL(10,2)
);

-- 10. Payments (1:1)
CREATE TABLE Payments (
    payment_id INT IDENTITY PRIMARY KEY,
    order_id INT UNIQUE FOREIGN KEY REFERENCES Orders(order_id),
    amount DECIMAL(10,2)
);

-- 11. Delivery_Assignments
CREATE TABLE Delivery_Assignments (
    assignment_id INT IDENTITY PRIMARY KEY,
    order_id INT UNIQUE FOREIGN KEY REFERENCES Orders(order_id),
    rider_id INT FOREIGN KEY REFERENCES Riders(rider_id),
    delivered_at DATETIME2 NULL
);

-- Analytical Queries
-- Top restaurant by revenue
SELECT TOP 1 r.name, SUM(o.total_amount) AS revenue
FROM Orders o JOIN Restaurants r ON o.restaurant_id = r.restaurant_id
GROUP BY r.name ORDER BY revenue DESC;

-- Average delivery time per rider
SELECT r.first_name, AVG(DATEDIFF(MINUTE, da.assigned_at, da.delivered_at)) AS avg_min
FROM Delivery_Assignments da JOIN Riders r ON da.rider_id = r.rider_id
WHERE da.delivered_at IS NOT NULL GROUP BY r.first_name;
